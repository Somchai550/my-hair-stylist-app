    <script>
        const startBtn = document.getElementById('startBtn');
        const stopBtn = document.getElementById('stopBtn');
        const video = document.getElementById('video');
        const faceCanvas = document.getElementById('faceCanvas');
        const loading = document.getElementById('loading');
        const hairstyleMenu = document.getElementById('hairstyleMenu');
        const hairstyleButtons = document.querySelectorAll('.hairstyle-btn');
        const ctx = faceCanvas.getContext('2d');
    
        let selectedHairstyle = null;
        let videoStream = null;
    
        const MODELS_URL = 'https://raw.githubusercontent.com/justadudewhohacks/face-api.js/master/weights/';
    
        // โหลดโมเดล AI
        Promise.all([
            faceapi.nets.tinyFaceDetector.loadFromUri(MODELS_URL),
            faceapi.nets.faceLandmark68Net.loadFromUri(MODELS_URL),
        ]).then(() => {
            loading.style.display = 'none';
            startBtn.style.display = 'block';
            console.log('Face detection models loaded!');
        }).catch(err => {
            console.error('Failed to load models:', err);
            loading.innerText = 'เกิดข้อผิดพลาดในการโหลดโมเดล';
        });
    
        // ฟังก์ชันสำหรับเริ่มใช้กล้อง
        startBtn.addEventListener('click', () => {
            navigator.mediaDevices.getUserMedia({ video: {} })
                .then(stream => {
                    video.srcObject = stream;
                    videoStream = stream;
                    video.style.display = 'block';
                    faceCanvas.style.display = 'block';
                    startBtn.style.display = 'none';
                    stopBtn.style.display = 'block';
                    hairstyleMenu.style.display = 'block';
                    video.addEventListener('play', () => {
                        setInterval(async () => {
                            const detections = await faceapi.detectSingleFace(video, new faceapi.TinyFaceDetectorOptions()).withFaceLandmarks();
                            ctx.clearRect(0, 0, faceCanvas.width, faceCanvas.height);
                            
                            if (detections) {
                                // ปรับขนาด canvas ให้เท่ากับ video
                                faceCanvas.width = video.videoWidth;
                                faceCanvas.height = video.videoHeight;
                                
                                // วาดภาพจากวิดีโอลงบน canvas
                                ctx.drawImage(video, 0, 0, faceCanvas.width, faceCanvas.height);
                                
                                // ถ้ามีทรงผมที่เลือกไว้ ให้วาดทับลงไป
                                if (selectedHairstyle) {
                                    const resizedDetections = faceapi.resizeResults(detections, { width: faceCanvas.width, height: faceCanvas.height });
                                    
                                    // โค้ดสำหรับวางทรงผมบนหัว (ส่วนนี้ต้องมีการปรับแต่งให้พอดี)
                                    // ตัวอย่างนี้จะวาดแบบง่ายๆ ลงไปตรงกลาง
                                    const headBox = resizedDetections.landmarks.getJawOutline()[8];
                                    const headX = headBox.x - (selectedHairstyle.width / 2);
                                    const headY = headBox.y - (selectedHairstyle.height * 0.8);
                                    
                                    ctx.globalAlpha = 0.8;
                                    ctx.drawImage(selectedHairstyle, 0, 0, faceCanvas.width, faceCanvas.height);
                                    ctx.globalAlpha = 1.0;
                                }
                            } else {
                                // ถ้าไม่พบใบหน้า ให้วาดภาพจากวิดีโอเฉยๆ
                                ctx.drawImage(video, 0, 0, faceCanvas.width, faceCanvas.height);
                            }
                        }, 100); // ตรวจจับทุกๆ 100 มิลลิวินาที (10 ครั้ง/วินาที)
                    });
                })
                .catch(err => {
                    console.error("Error accessing camera: ", err);
                    alert("ไม่สามารถเข้าถึงกล้องได้ กรุณาตรวจสอบสิทธิ์การใช้งาน");
                });
        });
        
        // ฟังก์ชันสำหรับหยุดใช้กล้อง
        stopBtn.addEventListener('click', () => {
            if (videoStream) {
                videoStream.getTracks().forEach(track => track.stop());
                videoStream = null;
                video.style.display = 'none';
                faceCanvas.style.display = 'none';
                startBtn.style.display = 'block';
                stopBtn.style.display = 'none';
                hairstyleMenu.style.display = 'none';
            }
        });
    
        // เพิ่ม Event Listener ให้กับปุ่มทรงผมทั้งหมด
        hairstyleButtons.forEach(button => {
            button.addEventListener('click', () => {
                const hairstyleUrl = button.getAttribute('data-url');
                const hairstyleImage = new Image();
                hairstyleImage.src = hairstyleUrl;
                hairstyleImage.onload = () => {
                    selectedHairstyle = hairstyleImage;
                };
            });
        });
    </script>
